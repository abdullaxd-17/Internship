/* â”€â”€ CS pin aliases â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */
#define CS1_PORT  GPIOA
#define CS1_PIN   GPIO_PIN_4
#define CS2_PORT  GPIOB
#define CS2_PIN   GPIO_PIN_0

/* â”€â”€ MAX31865 register addresses â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */
#define REG_CONFIG      0x00
#define REG_RTD_MSB     0x01
#define REG_RTD_LSB     0x02
#define REG_HF_MSB      0x03
#define REG_HF_LSB      0x04
#define REG_LF_MSB      0x05
#define REG_LF_LSB      0x06
#define REG_FAULT       0x07
#define WRITE_BIT       0x80   /* OR address with this to write */

/* â”€â”€ MAX31865 configuration byte â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */
/*
 *  Bit 7  VBIAS       1 = bias voltage ON  (must be ON for conversion)
 *  Bit 6  Auto/1shot  1 = continuous auto-convert
 *  Bit 5  1-shot      0 (unused in auto mode)
 *  Bit 4  3-wire      0 = 2-wire or 4-wire  |  1 = 3-wire
 *  Bit 3-2 FaultDetect 00 = no action
 *  Bit 1  FaultClear  0
 *  Bit 0  50/60 Hz    1 = 50 Hz filter  |  0 = 60 Hz filter
 *
 *  0xC1 = VBIAS=1, Auto=1, 2/4-wire, 50 Hz  â† suitable for most setups
 *  0xD1 = same but 3-wire
 *  0xC0 = VBIAS=1, Auto=1, 2/4-wire, 60 Hz  (change for 60 Hz countries)
 */
#define MAX31865_CONFIG   0xC1   /* 2/4-wire, 50 Hz filter, auto-convert */

/* â”€â”€ RTD sensor constants â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */
/*
 *  For PT100  : PT_R0 = 100.0,  PT_RREF = 430.0  (typical 430Î© reference)
 *  For PT1000 : PT_R0 = 1000.0, PT_RREF = 4300.0 (typical 4.3kÎ© reference)
 *  Change these to match your breakout board's reference resistor.

#define PT_R0       100.0f    /* RTD resistance at 0 Â°C (Î©) */
#define PT_RREF     430.0f    /* Reference resistor on breakout board (Î©) */

/* Callendar-Van Dusen coefficients (IEC 60751 standard) */
#define CVD_A    3.9083e-3f
#define CVD_B   -5.775e-7f
#define CVD_C   -4.183e-12f  /* used only for T < 0 Â°C */

/* â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â• main â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â• */

                t1_int, t1_frac);
        UART_SendString(buffer);
        
        HAL_Delay(2000);
    }
}

/* â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â• MAX31865 Driver Functions â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â• */

/* â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
 * MAX31865_Init
 *   Writes the configuration register to enable VBIAS and
 *   start continuous auto-conversion mode.
 * â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */
void MAX31865_Init(GPIO_TypeDef *cs_port, uint16_t cs_pin)
{
    char cs_name[10];
    if (cs_port == GPIOA && cs_pin == GPIO_PIN_4)
        strcpy(cs_name, "CS1/PA4");
    else if (cs_port == GPIOB && cs_pin == GPIO_PIN_0)
        strcpy(cs_name, "CS2/PB0");
    else
        strcpy(cs_name, "UNKNOWN");

    char msg[100];
    sprintf(msg, "[INIT] Initializing %s...\r\n", cs_name);
    UART_SendString(msg);

    /* Enable VBIAS and start auto-conversion */
    MAX31865_WriteReg(cs_port, cs_pin, REG_CONFIG, MAX31865_CONFIG);
    HAL_Delay(150);   /* Wait for VBIAS to stabilize and first conversion */
    
    /* Verify config was written */
    uint8_t config = MAX31865_ReadReg(cs_port, cs_pin, REG_CONFIG);
    sprintf(msg, "[INIT] %s Config reg = 0x%02X (expected 0xC1)\r\n", cs_name, config);
    UART_SendString(msg);

    if (config != MAX31865_CONFIG)
    {
        sprintf(msg, "[ERROR] %s config write FAILED! Check power, CS line, and SPI!\r\n", cs_name);
        UART_SendString(msg);
    }
}

/* â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
 * MAX31865_WriteReg
 *   Writes one byte to a MAX31865 register.
 *   Address byte is OR'd with 0x80 to signal a write operation.
 * â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */
void MAX31865_WriteReg(GPIO_TypeDef *cs_port, uint16_t cs_pin,
                       uint8_t reg, uint8_t value)
{
    uint8_t tx[2] = { (uint8_t)(reg | WRITE_BIT), value };
    
    char debug_buf[100];
    sprintf(debug_buf, "[SPI_WRITE] CS toggling... Reg=0x%02X, Val=0x%02X\r\n", reg, value);
    UART_SendString(debug_buf);

    HAL_GPIO_WritePin(cs_port, cs_pin, GPIO_PIN_RESET);  /* CS LOW */
    HAL_Delay(1);
    
    HAL_StatusTypeDef status = HAL_SPI_Transmit(&hspi1, tx, 2, 50);
    if (status != HAL_OK)
        UART_SendString("[ERROR] SPI_Transmit failed!\r\n");
    
    HAL_Delay(1);
    HAL_GPIO_WritePin(cs_port, cs_pin, GPIO_PIN_SET);    /* CS HIGH */
    
    sprintf(debug_buf, "[SPI_WRITE] Done\r\n");
    UART_SendString(debug_buf);
}

/* â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
 * MAX31865_ReadReg
 *   Reads one byte from a MAX31865 register.
 * â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */
uint8_t MAX31865_ReadReg(GPIO_TypeDef *cs_port, uint16_t cs_pin, uint8_t reg)
{
    uint8_t addr = reg & 0x7F;   /* read: MSB = 0 */
    uint8_t rx   = 0;

    HAL_GPIO_WritePin(cs_port, cs_pin, GPIO_PIN_RESET);
    HAL_SPI_Transmit(&hspi1, &addr, 1, 50);
    HAL_SPI_Receive (&hspi1, &rx,   1, 50);
    HAL_GPIO_WritePin(cs_port, cs_pin, GPIO_PIN_SET);

    return rx;
}

/* â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
 * MAX31865_ReadTemperature
 *   Reads the 15-bit RTD ADC result, converts to resistance,
 *   then to temperature using Callendar-Van Dusen.
 *
 *   RTD register layout (two bytes, address 0x01-0x02):
 *   [15:1]  15-bit ADC value  (MSB first)
 *   [0]     Fault flag        (1 = fault present)
 *
 *   Returns -999.0f on fault.
 * â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */
float MAX31865_ReadTemperature(GPIO_TypeDef *cs_port, uint16_t cs_pin)
{
    /* Burst-read RTD MSB and LSB in one transaction */
    uint8_t addr   = REG_RTD_MSB & 0x7F;
    uint8_t rx[2]  = {0, 0};

    char cs_name[10];
    if (cs_port == GPIOA && cs_pin == GPIO_PIN_4)
        strcpy(cs_name, "CS1/PA4");
    else if (cs_port == GPIOB && cs_pin == GPIO_PIN_0)
        strcpy(cs_name, "CS2/PB0");
    else
        strcpy(cs_name, "UNKNOWN");

    HAL_GPIO_WritePin(cs_port, cs_pin, GPIO_PIN_RESET);
    HAL_SPI_Transmit(&hspi1, &addr,  1, 50);
    HAL_SPI_Receive (&hspi1,  rx,    2, 50);
    HAL_GPIO_WritePin(cs_port, cs_pin, GPIO_PIN_SET);

    /* Bit 0 of LSB is the fault flag */
    if (rx[1] & 0x01)
    {
        /* Read fault register for diagnosis */
        uint8_t fault = MAX31865_ReadReg(cs_port, cs_pin, REG_FAULT);
        char buf[100];
        sprintf(buf, "FAULT DETECTED! Reg=0x%02X\r\n", fault);
        UART_SendString(buf);
        return -999.0f;
    }

    /* Combine bytes into 15-bit ADC value (drop fault bit) */
    uint16_t adc_code = (uint16_t)(((uint16_t)rx[0] << 8) | rx[1]) >> 1;

    /* Convert ADC code â†’ RTD resistance
     *   R_rtd = adc_code * R_ref / 32768
     *   32768 = 2^15 (full-scale for 15-bit ADC)                        */
    float r_rtd = (float)adc_code * PT_RREF / 32768.0f;
    
    /* DEBUG: Print raw values */
    char debug_buf[100];
    int adc_int = (int)adc_code;
    int r_rtd_int = (int)r_rtd;
    int r_rtd_frac = (int)((r_rtd - r_rtd_int) * 100);
    sprintf(debug_buf, "[DEBUG] %s ADC=0x%04X (%d) | R_rtd=%d.%02d Ohm\r\n", 
            cs_name, adc_code, adc_int, r_rtd_int, r_rtd_frac);
    UART_SendString(debug_buf);

    /* Sensor malfunction detection */
    if (adc_code < 500)
    {
        sprintf(debug_buf, "[WARN] %s ADC very low! Check: 1) PT100 connected? 2) CS toggling? 3) Power supply?\r\n", cs_name);
        UART_SendString(debug_buf);
    }
    if (adc_code > 32500)
    {
        sprintf(debug_buf, "[WARN] %s ADC very high! Check: 1) Sensor disconnected? 2) Reference resistor OK?\r\n", cs_name);
        UART_SendString(debug_buf);
    }

    return RTD_ToTemperature(r_rtd);
}

/* â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
 * RTD_ToTemperature
 *   Converts RTD resistance to Â°C using the Callendar-Van Dusen equation.
 *
 *   For T â‰¥ 0 Â°C:  Rt = R0(1 + AÂ·T + BÂ·TÂ²)
 *     Solving quadratic:  T = (âˆ’A + âˆš(AÂ²âˆ’4B(1âˆ’Rt/R0))) / (2B)
 *
 *   For T < 0 Â°C:  Exact CVD has a 4th-degree term (coefficient C).
 *     Below uses a linearised approximation valid to about âˆ’50 Â°C.
 *     For accuracy below âˆ’50 Â°C replace with a Newton-Raphson iteration.
 * â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */
float RTD_ToTemperature(float r_rtd)
{
    float z1, z2, z3, z4, temp;

    z1 = -CVD_A;
    z2 = CVD_A * CVD_A - 4.0f * CVD_B;
    z3 = (4.0f * CVD_B) / PT_R0;
    z4 = 2.0f * CVD_B;

    temp = z2 + z3 * r_rtd;
    temp = (sqrtf(temp) + z1) / z4;

    if (temp >= 0.0f)
        return temp;

    /* Below 0 Â°C: use polynomial approximation (accurate to ~Â±0.5 Â°C) */
    float rpoly = r_rtd;
    temp  = -242.02f;
    temp +=   2.2228f  * rpoly;
    rpoly *= r_rtd;   /* rÂ² */
    temp +=   2.5859e-3f * rpoly;
    rpoly *= r_rtd;   /* rÂ³ */
    temp -=   4.8260e-6f * rpoly;
    rpoly *= r_rtd;   /* râ´ */
    temp -=   2.8183e-8f * rpoly;
    rpoly *= r_rtd;   /* râµ */
    temp +=   1.5243e-10f * rpoly;

    return temp;
}
